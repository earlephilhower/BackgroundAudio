225 virtual size_t write(const uint8_t *buffer, size_t size) override {
226 size_t written = 0;
227
228 if (!_running) {
229 return 0;
230 }
231 size_t words = size / sizeof(uint32_t);
232 while (words) {
233 AudioBuffer **volatile p = (AudioBuffer * *volatile)&_empty;
234 if (!*p) {
235 break;
236 }
237 size_t availToWriteThisBuff = _wordsPerBuffer - _userOff;
238 size_t toWrite = std::min(availToWriteThisBuff, words);
239 memcpy(&((*p)->buff[_userOff]), buffer, toWrite * sizeof(uint32_t));
240 buffer += toWrite * sizeof(uint32_t);
241 written += toWrite;
242 _userOff += toWrite;
243 words -= toWrite;
244 if (_userOff == _wordsPerBuffer) {
245 _addToList(&_filled, _takeFromList(p));
246 _userOff = 0;
247 }
248 }
249 return written * sizeof(uint32_t);
250 }

275 void _addToList(AudioBuffer **list, AudioBuffer *element) {
276 noInterrupts();
277 // Find end of list, if any
278 while ((*list) && ((*list)->next != nullptr)) {

305 uint32_t *getResampledBuffer() {
306 // Prime the pump when we're all out of frames
307 if (!_filled && _cb) {
308 _cb(_cbData);
309 }
310
311 if (!_filled) {
312 for (size_t i = 0; i < _outputBufferWords; i++) {
313 _outputBuffer[i] = _silenceSample;
314 _underflow = true;
315 }
316 return _outputBuffer; // Nothing to send in but the sound of silence...
317 }
318
319 uint32_t *p = _outputBuffer;
320 if (_outputRate == _inputRate) {
321 for (size_t i = 0; i < _outputBufferWords; i++) {
322 if (_readOff == _wordsPerBuffer) {
323 _addToList(&_empty, _takeFromList(&_filled));
324 _readOff = 0;
325 if (_cb) {
326 _cb(_cbData);
327 }
328 }
329 if (!_filled) {
330 _underflow = true;
331 }
332 *(p++) = _filled ? _filled->buff[_readOff++] : _silenceSample;
333 }
334 } else {
335 // _readOff will be a 16p16 fixed point
336 for (size_t i = 0; i < _outputBufferWords; i++) {
337 while (_readOff >= _wordsPerBuffer << 16) {
338 _addToList(&_empty, _takeFromList(&_filled));
339 _readOff -= _wordsPerBuffer << 16;
340 if (_cb) {
341 _cb(_cbData);
342 }
343 }
344 if (!_filled) {
345 _underflow = true;
346 }
347 *(p++) = _filled ? _filled->buff[_readOff >> 16] : _silenceSample;
348 if (_filled) {
349 _readOff += _resample16p16;
350 }
351 }
352 }
353 return _outputBuffer;
354 }

415 BackgroundAudioMixerInput *add() {
416 if (_running) {
417 return nullptr;
418 }
419 auto x = new BackgroundAudioMixerInput(_outRate, _outWords);
420 _input.push_back(x);
421 return x;

434 bool begin() {
435 if (_running) {
436 return false;
437 }
438
439 // We will use natural frame size to minimize mismatch
440 _out->setBuffers(4, _outWords); // Framelen is in samples, but buffers takes words, which means 2 samples per word so we're good!
441 _out->onTransmit(&_cb, (void *)this); // The pump we will use to generate our audio
442 _out->setBitsPerSample(16);
443 _out->setStereo(true);
444 _out->setFrequency(_outRate);
445 _out->begin();
446
447 // Stuff with silence to start
448 uint32_t zeros[32] __attribute__((aligned(4))) = {};
449 while (_out->availableForWrite() > 32) {
450 _out->write((uint8_t *)zeros, sizeof(zeros));
451 }
452
453 _running = true;
454
455 return true;
456 }

458private:
462 static void _cb(void *ptr) {
463 ((BackgroundAudioMixer *)ptr)->pump();
464 }

469 void generateOneFrame() {
470 // Collect all the input leg buffers
471 int16_t *leg[_input.size()];
472 for (size_t i = 0; i < _input.size(); i++) {
473 leg[i] = (int16_t *)_input[i]->getResampledBuffer();
474 }
475
476 // Sum them up with saturating arithmetic
477 for (size_t i = 0; i < _outWords * 2; i++) {
478 int32_t sum = 0;
479 for (size_t j = 0; j < _input.size(); j++) {
480 sum += leg[j][i];
481 }
482 if (sum > 32767) {
483 sum = 32767;
484 } else if (sum < -32767) {
485 sum = -32767;
486 }
487 _outBuff[i] = (int16_t)sum;
488 }
489 }
490
494 void pump() {
495 while (_out->availableForWrite() >= (int)_outWords) {
496 generateOneFrame();
497 _out->write((uint8_t *)_outBuff, _outWords * 4);
498 }
499 }

501 bool _running;
502 AudioOutputBase *_out;
503 int _outRate;
504 int16_t _outBuff[_outWords * 2];
505 std::vector<BackgroundAudioMixerInput *> _input;
