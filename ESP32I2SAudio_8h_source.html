157 bool begin() override {
158 if (_running) {
159 return false;
160 }
161
162 // Make a new channel of the requested buffers (which may be ignored by the IDF!)
163 i2s_chan_config_t chan_cfg = I2S_CHANNEL_DEFAULT_CONFIG(I2S_NUM_AUTO, I2S_ROLE_MASTER);
164 chan_cfg.dma_desc_num = _buffers;
165 chan_cfg.dma_frame_num = _bufferWords * 4;
166 i2s_new_channel(&chan_cfg, &_tx_handle, nullptr);
167
168 i2s_std_config_t std_cfg = {
169 .clk_cfg = I2S_STD_CLK_DEFAULT_CONFIG(_sampleRate),
170 .slot_cfg = I2S_STD_MSB_SLOT_DEFAULT_CONFIG(I2S_DATA_BIT_WIDTH_16BIT, I2S_SLOT_MODE_STEREO),
171 .gpio_cfg = {
172 .mclk = _mclk < 0 ? I2S_GPIO_UNUSED : (gpio_num_t)_mclk,
173 .bclk = (gpio_num_t)_bclk,
174 .ws = (gpio_num_t)_ws,
175 .dout = (gpio_num_t)_dout,
176 .din = I2S_GPIO_UNUSED,
177 .invert_flags = {
178 .mclk_inv = _mclkInv,
179 .bclk_inv = _bclkInv,
180 .ws_inv = _wsInv,
181 },
182 },
183 };
184 i2s_channel_init_std_mode(_tx_handle, &std_cfg);
185
186 // Prefill silence and calculate how bug we really have
187 int16_t a[2] = {0, 0};
188 size_t written = 0;
189 do {
190 i2s_channel_preload_data(_tx_handle, (void*)a, sizeof(a), &written);
191 _totalAvailable += written;
192 } while (written);
193
194 // The IRQ callbacks which will just trigger the playback task
195 i2s_event_callbacks_t _cbs = {
196 .on_recv = nullptr,
197 .on_recv_q_ovf = nullptr,
198 .on_sent = _onSent,
199 .on_send_q_ovf = _onSentUnder
200 };
201 i2s_channel_register_event_callback(_tx_handle, &_cbs, (void *)this);
202 xTaskCreate(_taskShim, "BackgroundAudioI2S", 4096, (void*)this, 2, &_taskHandle);
203 _running = ESP_OK == i2s_channel_enable(_tx_handle);
204 return _running;
205 }

235 void _backgroundTask() {
236 while (true) {
237 // Pause the task until notification comes in from the IRQ callbacks
238 uint32_t ulNotifiedValue;
239 xTaskNotifyWait(0, ULONG_MAX, &ulNotifiedValue, portMAX_DELAY);
240 // One DMA transfer == one frame
241 _frames++;
242 // Use the notification value to transmit how much data was in the DMA block. When negative this was an underflow notification
243 int32_t size = (int32_t)ulNotifiedValue;
244 if (size < 0) {
245 _underflows++;
246 size = -size;
247 }
248 // Track the amount believed available, with some sanity checking
249 _available += size;
250 if (_available > _totalAvailable) {
251 _available = _totalAvailable;
252 }
253 // Callback used from within a normal FreeRTOS task, so it can be slow and/or write I2S
254 if (_cb) {
255 _cb(_cbData);
256 }
257 }
258 }

293 bool _onSentCB(i2s_chan_handle_t handle, i2s_event_data_t *event, bool underflow = false) {
294 BaseType_t xHigherPriorityTaskWoken;
295 xHigherPriorityTaskWoken = pdFALSE;
296 if (_taskHandle) {
297 _irqs++;
298 if (underflow) {
299 _underflowed = true;
300 }
301 xTaskNotifyFromISR(_taskHandle, event->size * (underflow ? -1 : 1), eSetValueWithOverwrite, &xHigherPriorityTaskWoken);
302 }
303 return (bool)xHigherPriorityTaskWoken;
304 }
