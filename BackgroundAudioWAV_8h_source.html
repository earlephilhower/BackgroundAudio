266 void generateOneFrame() {
267 int16_t *out = (int16_t *)_outSample;
268 int16_t *end = &_outSample[framelen * 2];
269 while (out < end) {
270 // If we accumulate too large of a shift, actually do the shift so more space for writer
271 if (_accumShift > _ib.size() / 2) {
272 _ib.shiftUp(_accumShift);
273 _accumShift = 0;
274 _shifts++;
275 }
276
277 const unsigned char *b = _ib.buffer() + _accumShift;
278 int avail = _ib.available() - _accumShift;
279 if (avail <= 0) {
280 goto underflow;
281 }
282 if (_dataSkipped) {
283 int toSkip = std::min(_dataSkipped, avail);
284 _accumShift += toSkip;
285 _dataSkipped -= toSkip;
286 continue;
287 }
288 if (_dataRemaining > 0) {
289 if ((_channels == 1) && (_bps == 8)) {
290 if (avail < 1) {
291 goto underflow;
292 }
293 int16_t l = b[0]; // 0-255 unsigned
294 l -= 128; // -128...127 signed
295 l <<= 8; // -32K..32K signed
296 *out++ = l;
297 *out++ = l;
298 _accumShift++;
299 } else if ((_channels == 2) && (_bps == 8)) {
300 if (avail < 2) {
301 goto underflow;
302 }
303 int16_t l = b[0]; // 0-255 unsigned
304 l -= 128; // -128...127 signed
305 l <<= 8; // -32K..32K signed
306 *out++ = l;
307 int16_t r = b[1]; // 0-255 unsigned
308 r -= 128; // -128...127 signed
309 r <<= 8; // -32K..32K signed
310 *out++ = r;
311 _accumShift += 2;
312 } else if ((_channels == 1) && (_bps == 16)) {
313 if (avail < 2) {
314 goto underflow;
315 }
316 int16_t l = b[0] | (b[1] << 8);
317 *out++ = l;
318 *out++ = l;
319 _accumShift += 2;
320 } else { // 16b, stereo
321 if (avail < 4) {
322 goto underflow;
323 }
324 int16_t l = b[0] | (b[1] << 8);
325 *out++ = l;
326 int16_t r = b[2] | (b[3] << 8);
327 *out++ = r;
328 _accumShift += 4;
329 }
330 _dataRemaining--;
331 if (!_dataRemaining) {
332 _seenDATA = false;
333 }
334 continue;
335 } else {
336 if (avail < 4) { // Need 4 bytes to get a header type
337 goto underflow;
338 }
339 if (!memcmp(b, "RIFF", 4)) {
340 if (avail < 12) {
341 goto underflow;
342 }
343 if (!memcmp(b + 8, "WAVE", 4)) {
344 // This is a RIFF header. We will ignore the total size
345 _seenRIFF = true;
346 _accumShift += 12; // Skip it
347 continue;
348 }
349 _accumShift++; // Error, shift along
350 continue;
351 } else if (_seenRIFF && !_seenFMT && !memcmp(b, "fmt ", 4)) {
352 if (avail < 24) {
353 goto underflow;
354 }
355 if ((b[4] != 16) || b[5] || b[6] || b[7] || (b[8] != 1) || b[9]) {
356 // Length for format not PCM, ignore this
357 _errors++;
358 _seenRIFF = false;
359 _accumShift++;
360 continue;
361 }
362 _channels = b[10] | (b[11] << 8);
363 _sampleRate = b[12] | (b[13] << 8) | (b[14] << 16) | (b[15] << 24);
364 _bps = b[22] | (b[23] << 8);
365 if ((_channels == 0) || (_channels > 2) || !((_bps == 8) || (_bps == 16)) || (_sampleRate < 4000) || (_sampleRate > 48000)) {
366 // Invalid config, we can't play
367 _errors++;
368 _sampleRate = 0;
369 _seenRIFF = false;
370 _accumShift++;
371 continue;
372 }
373 _accumShift += 24;
374 _seenFMT = true;
375 } else if (_seenRIFF && _seenFMT && !_seenDATA && !memcmp(b, "data", 4)) {
376 if (avail < 8) {
377 goto underflow;
378 }
379 _dataRemaining = b[4] | (b[5] << 8) | (b[6] << 16) | (b[7] << 24); // in bytes
380 _dataRemaining /= _bps / 8;
381 _dataRemaining /= _channels; // Now in samples
382 _seenDATA = true;
383 _seenRIFF = false;
384 _seenFMT = false;
385 _accumShift += 8; // Jump to the data
386 continue;
387 } else {
388 if (avail < 8) {
389 goto underflow;
390 }
391 _dataSkipped = b[4] | (b[5] << 8) | (b[6] << 16) | (b[7] << 24); // in bytes
392 _accumShift += 8;
393 // We'll just skip this chunk before processing
394 continue;
395 }
396 }
397underflow:
398 _underflows++;
399 if (_accumShift) {
400 _ib.shiftUp(_accumShift);
401 _accumShift = 0;
402 _shifts++;
403 }
404 // Because we're in IRQ and there's not enough data, we know no more can come it so just 0-fill remainder in tight loop
405 while (out < end) {
406 *out++ = 0;
407 *out++ = 0;
408 }
409 continue;
410 } // while(out < end-of-outsamples)
411
412 ApplyGain(_outSample, framelen * 2, _gain);
413 }

417 void pump() {
418 while (_out->availableForWrite() >= (int)framelen) {
419 if (_paused) {
420 bzero((uint8_t *)_outSample, framelen * 2 * sizeof(int16_t));
421 } else {
422 generateOneFrame();
423 if (_sampleRate) {
424 _out->setFrequency(_sampleRate);
425 }
426 }
427 _out->write((uint8_t *)_outSample, framelen * 2 * sizeof(int16_t));
428 }
429 }
